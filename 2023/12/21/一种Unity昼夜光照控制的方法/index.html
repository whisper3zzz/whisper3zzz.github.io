一种Unity昼夜光照控制的方法

2023/12/21 教程 Unity

刚接个活，写了个脚本控制场景简单的光照昼夜变化，记录一下。

using UnityEngine;

public class DayNightCycle : MonoBehaviour
{
    public Light sun;
    public Light lightEmit;
    public Light[] indoorLights;

    public float dayDuration = 60f; // 一天的持续时间（秒）

        void Update()
    {
        // 计算当前时间（0到1之间）
        float currentTime = Mathf.Repeat(Time.time / dayDuration, 1f);

        // 根据当前时间设置不同的状态
        if (currentTime < 0.5f)
            {
            // 白天
                SetDaytime();
            }
            else if (currentTime < 0.75f)
        {
                // 傍晚
                SetEvening();
        }
            else
            {
                // 夜晚
                SetNight();
        }

                // 根据当前时间旋转太阳
            sun.transform.eulerAngles = new Vector3(currentTime * 360f - 90f, 30f, 0f);
        }

        void SetDaytime()
        {
                // 白天状态
            sun.intensity = 1f;
            lightEmit.intensity = 1f;

                // 关闭室内灯光
            foreach (Light indoorLight in indoorLights)
                {
                    indoorLight.intensity = 0f;
                }
        }

        void SetEvening()
        {
                // 傍晚状态
        sun.intensity = 0.5f;
        lightEmit.intensity = 0.5f;

            // 部分室内灯光打开
            foreach (Light indoorLight in indoorLights)
            {
                indoorLight.intensity = 0.5f;
            }
        }

        void SetNight()
        {
            // 夜晚状态
            sun.intensity = 0f;
            lightEmit.intensity = 0f;

            // 所有室内灯光打开
            foreach (Light indoorLight in indoorLights)
            {
                indoorLight.intensity = 1f;
            }
        }
    }
